Module bitrab.tui.orchestrator

TUI-aware pipeline orchestrator.

Runs jobs in parallel via ProcessPoolExecutor, routing output through a multiprocessing.Manager().Queue() so the Textual app can display per-job logs.

For CI mode (no TUI), jobs write to per-job log files and the files are printed to stdout after each stage completes.

class QueueWriter:
    """File-like object that puts text into a multiprocessing Manager Queue.

    Must be at module level (not nested) to be picklable under spawn context.
    """

    def __init__(self, queue: Any, job_name: str) -> None:
        self._queue = queue
        self._job_name = job_name

    def write(self, s: str) -> None:
        """Write text to the queue."""
        if s:
            self._queue.put((self._job_name, s))

    def flush(self) -> None:
        """No-op flush to satisfy IO protocol."""

class TUIOrchestrator:
    """Pipeline orchestrator that feeds output to the Textual TUI or CI log files.

    Attributes:
        job_executor: The JobExecutor used to run individual jobs.
        maximum_degree_of_parallelism: Max parallel jobs per stage.
    """

    def __init__(
        self,
        job_executor: JobExecutor,
        maximum_degree_of_parallelism: int | None = None,
        mp_ctx: Any = None,
    ) -> None:
        self.job_executor = job_executor
        cpu_cnt = os.cpu_count() or 1
        self.maximum_degree_of_parallelism = cpu_cnt if maximum_degree_of_parallelism is None else max(1, maximum_degree_of_parallelism)
        if mp_ctx is None:
            if sys.platform == "win32":
                mp_ctx = mp.get_context("spawn")
            else:
                mp_ctx = mp.get_context("spawn")
        self._mp_ctx = mp_ctx

        # Cancel/control state
        self._cancel_event: threading.Event = threading.Event()
        # job_name -> worker OS PID (populated via Manager().dict() in parallel path)
        self._worker_pids: dict[str, int] = {}

    def reset(self) -> None:
        """Reset orchestrator state for a fresh pipeline run."""
        self._cancel_event.clear()
        self._worker_pids = {}

    def cancel_pipeline(self) -> None:
        """Signal the pipeline to stop after the current stage completes."""
        self._cancel_event.set()

    def cancel_job(self, job_name: str) -> None:
        """Best-effort: send SIGTERM/taskkill to the worker process running job_name.

        Note: on Windows, taskkill /F kills the entire process tree immediately,
        so after_script will NOT run for the cancelled job.
        """
        pid = self._worker_pids.get(job_name)
        if pid is None:
            return
        try:
            if sys.platform == "win32":
                subprocess.run(["taskkill", "/F", "/PID", str(pid)], check=False, capture_output=True)  # nosec
            else:
                os.kill(pid, signal.SIGTERM)
        except (ProcessLookupError, PermissionError, OSError):
            pass  # process already exited

    def run_job_inline(self, job: JobConfig, app: PipelineApp) -> None:
        """Run a single job inline in the calling thread (for restart-job).

        Must be called from a background thread (not the event loop thread).
        Creates its own ephemeral Manager and Queue.
        """
        from bitrab.tui.app import JobOutput, JobStatusChanged

        mgr = self._mp_ctx.Manager()
        output_queue = mgr.Queue()
        job_dir = self.job_executor.project_dir / ".bitrab" / _sanitize_job_name(job.name)
        job_dir.mkdir(parents=True, exist_ok=True)
        app.call_from_thread(app.post_message, JobStatusChanged(job.name, "running"))
        writer = QueueWriter(output_queue, job.name)
        try:
            self.job_executor.execute_job(job, job_dir=job_dir, output_writer=writer)
            app.call_from_thread(app.post_message, JobStatusChanged(job.name, "success"))
        except Exception:
            app.call_from_thread(app.post_message, JobStatusChanged(job.name, "failed"))
        finally:
            self._drain_queue_sync(output_queue, app, JobOutput)
            mgr.shutdown()

    def execute_pipeline_tui(self, pipeline: PipelineConfig, app: PipelineApp) -> None:
        """Execute pipeline with live output routed to Textual TUI.

        Called from a background thread (via app.run_worker(thread=True)).
        Uses call_from_thread to post messages to the event loop.
        """
        from bitrab.tui.app import JobOutput, JobStatusChanged

        jobs_by_stage = self._organize_jobs_by_stage(pipeline)

        # Create Manager once for the whole pipeline run
        mgr = self._mp_ctx.Manager()
        output_queue = mgr.Queue()
        worker_pids = mgr.dict()
        self._worker_pids = worker_pids  # type: ignore[assignment]

        try:
            for stage in pipeline.stages:
                # Check for cancellation before starting each new stage
                if self._cancel_event.is_set():
                    app.call_from_thread(app.on_pipeline_cancelled)
                    return

                stage_jobs = jobs_by_stage.get(stage, [])
                if not stage_jobs:
                    continue

                app.call_from_thread(app.update_stage_status, stage, len(stage_jobs))

                if self.maximum_degree_of_parallelism == 1 or len(stage_jobs) == 1:
                    # Single-job path: run inline in this thread, still use queue
                    for job in stage_jobs:
                        job_dir = self.job_executor.project_dir / ".bitrab" / _sanitize_job_name(job.name)
                        job_dir.mkdir(parents=True, exist_ok=True)
                        app.call_from_thread(app.post_message, JobStatusChanged(job.name, "running"))
                        writer = QueueWriter(output_queue, job.name)
                        try:
                            self.job_executor.execute_job(job, job_dir=job_dir, output_writer=writer)
                            app.call_from_thread(app.post_message, JobStatusChanged(job.name, "success"))
                        except Exception as exc:
                            app.call_from_thread(app.post_message, JobStatusChanged(job.name, "failed"))
                            raise exc
                        finally:
                            # drain any queued output
                            self._drain_queue_sync(output_queue, app, JobOutput)
                else:
                    active_jobs: set[str] = set()
                    failures: list[tuple[JobConfig, BaseException]] = []

                    with ProcessPoolExecutor(
                        max_workers=self.maximum_degree_of_parallelism,
                        mp_context=self._mp_ctx,
                    ) as pool:
                        futures = {}
                        for job in stage_jobs:
                            job_dir = self.job_executor.project_dir / ".bitrab" / _sanitize_job_name(job.name)
                            job_dir.mkdir(parents=True, exist_ok=True)
                            app.call_from_thread(app.post_message, JobStatusChanged(job.name, "running"))
                            active_jobs.add(job.name)
                            futures[pool.submit(_run_single_job_queued, job, self.job_executor, job_dir, output_queue, worker_pids)] = job

                        # Drain queue while futures run
                        while active_jobs:
                            try:
                                job_name, text = output_queue.get(timeout=0.05)
                                if text is None:
                                    active_jobs.discard(job_name)
                                else:
                                    app.call_from_thread(app.post_message, JobOutput(job_name, text))
                            except Empty:
                                if all(f.done() for f in futures):
                                    break

                        # Collect results
                        for fut in as_completed(futures):
                            job = futures[fut]
                            try:
                                history = fut.result()
                                self.job_executor.job_history.extend(history)
                                app.call_from_thread(app.post_message, JobStatusChanged(job.name, "success"))
                            except BaseException as exc:
                                failures.append((job, exc))
                                app.call_from_thread(app.post_message, JobStatusChanged(job.name, "failed"))

                    if failures:
                        raise failures[0][1]

            # After all stages complete, check if cancelled during final stage
            if self._cancel_event.is_set():
                app.call_from_thread(app.on_pipeline_cancelled)
            else:
                app.call_from_thread(app.on_pipeline_complete, True)
        except Exception as exc:
            app.call_from_thread(app.on_pipeline_complete, False)
            raise exc
        finally:
            mgr.shutdown()

    def execute_pipeline_ci(self, pipeline: PipelineConfig) -> None:
        """Execute pipeline in CI mode: jobs write to files, printed when done.

        No TUI. Jobs still run in parallel per stage for speed.
        Output files written to .bitrab/<job_name>/output.log
        """
        print("🚀 Starting pipeline (CI mode - parallel jobs, sequential stages)")
        print(f"📋 Stages: {', '.join(pipeline.stages)}")

        jobs_by_stage = self._organize_jobs_by_stage(pipeline)

        for stage in pipeline.stages:
            stage_jobs = jobs_by_stage.get(stage, [])
            if not stage_jobs:
                print(f"⏭️  Skipping empty stage: {stage}")
                continue

            print(f"\n🎯 Stage: {stage} ({len(stage_jobs)} job(s) running in parallel)")

            log_paths: dict[str, Path | None] = {}
            failures: list[tuple[JobConfig, BaseException]] = []

            if self.maximum_degree_of_parallelism == 1 or len(stage_jobs) == 1:
                for job in stage_jobs:
                    job_dir = self.job_executor.project_dir / ".bitrab" / _sanitize_job_name(job.name)
                    log_path = job_dir / "output.log"
                    log_paths[job.name] = log_path
                    job_dir.mkdir(parents=True, exist_ok=True)
                    print(f"  ▶ {job.name}")
                    try:
                        with open(log_path, "w", encoding="utf-8") as fh:
                            self.job_executor.execute_job(job, job_dir=job_dir, output_writer=fh)
                    except Exception as exc:
                        failures.append((job, exc))
            else:
                with ProcessPoolExecutor(
                    max_workers=self.maximum_degree_of_parallelism,
                    mp_context=self._mp_ctx,
                ) as pool:
                    futures = {}
                    for job in stage_jobs:
                        job_dir = self.job_executor.project_dir / ".bitrab" / _sanitize_job_name(job.name)
                        log_path = job_dir / "output.log"
                        log_paths[job.name] = log_path
                        print(f"  ▶ {job.name}")
                        futures[pool.submit(_run_single_job_file, job, self.job_executor, job_dir, log_path)] = job

                    for fut in as_completed(futures):
                        job = futures[fut]
                        try:
                            history = fut.result()
                            self.job_executor.job_history.extend(history)
                        except BaseException as exc:
                            failures.append((job, exc))

            # Print all logs after stage completes
            for job in stage_jobs:
                job_log: Path | None = log_paths.get(job.name)
                status = "✅" if not any(j.name == job.name for j, _ in failures) else "❌"
                print(f"\n{'=' * 60}")
                print(f"{status} Job: {job.name} (stage: {stage})")
                print(f"{'=' * 60}")
                if job_log and job_log.exists():
                    print(job_log.read_text(encoding="utf-8"))
                else:
                    print("(no output)")

            if failures:
                print(f"\n🛑 Stage {stage} failed. Stopping pipeline.")
                raise failures[0][1]

        print("\n🎉 Pipeline completed successfully!")

    def _drain_queue_sync(self, queue: Any, app: PipelineApp, job_output_cls: Any) -> None:
        """Drain any remaining items from queue after single-job inline execution."""
        while True:
            try:
                job_name, text = queue.get_nowait()
                if text is not None:
                    app.call_from_thread(app.post_message, job_output_cls(job_name, text))
            except Empty:
                break

    def _organize_jobs_by_stage(self, pipeline: PipelineConfig) -> dict[str, list[JobConfig]]:
        """Organize jobs by stage."""
        jobs_by_stage: dict[str, list[JobConfig]] = {}
        for job in pipeline.jobs:
            jobs_by_stage.setdefault(job.stage, []).append(job)
        return jobs_by_stage
